	name VARCHAR(50) UNIQUE NOT NULL
);

-- Permissions
CREATE TABLE permissions (
	id SERIAL PRIMARY KEY,
	name VARCHAR(50) UNIQUE NOT NULL
);

-- Role - permissions
CREATE TABLE role_permissions (
	role_id INT REFERENCES roles(id) ON DELETE CASCADE,
	permission_id INT REFERENCES permissions(id) ON DELETE CASCADE,
	PRIMARY KEY(role_id, permission_id)
);

--  users
CREATE TABLE users (
	id SERIAL PRIMARY KEY,
	email VARCHAR(255) UNIQUE NOT NULL,
	password VARCHAR(255) NOT NULL,
	role_id INT NOT NULL,
	created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
	updated_at TIMESTAMP,
	deleted_at TIMESTAMP DEFAULT NULL,
	CONSTRAINT fk_users FOREIGN KEY(role_id) REFERENCES roles(id) ON DELETE CASCADE
);

-- Refresh Tokens
CREATE TABLE refresh_tokens (
	id SERIAL PRIMARY KEY,
	user_id INT REFERENCES users(id) ON DELETE CASCADE,
	token TEXT NOT NULL,
	expires_at TIMESTAMP NOT NULL,
	created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Employee Table
CREATE TABLE employees(
	id BIGSERIAL PRIMARY KEY,
	emp_code VARCHAR(20) NOT NULL,
	user_id INT NOT NULL,
	full_name VARCHAR(255) NOT NULL,
	email VARCHAR(255) UNIQUE NOT NULL,
	phone VARCHAR(15) UNIQUE NOT NULL,
	employment_category_id BIGINT NOT NULL,
	dept_id BIGINT NOT NULL,
	desg_id BIGINT NOT NULL,
	date_of_joining DATE NOT NULL,

	created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
	updated_at TIMESTAMP DEFAULT NULL,
	deleted_at TIMESTAMP DEFAULT NULL,
	CONSTRAINT fk_dept FOREIGN KEY(dept_id) REFERENCES departments(id) ON DELETE CASCADE,
	CONSTRAINT fk_desg FOREIGN KEY(desg_id) REFERENCES designations(id) ON DELETE CASCADE,
	CONSTRAINT fk_user FOREIGN KEY(user_id) REFERENCES users(id) ON DELETE CASCADE,
	CONSTRAINT fk_emp_category FOREIGN KEY(employment_category_id) REFERENCES employment_categories(id) ON DELETE CASCADE
);

CREATE TABLE employee_leave_balance (
	balance_id BIGSERIAL PRIMARY KEY,
	employee_id BIGINT NOT NULL,
	leave_id BIGINT NOT NULL,
	opening_balance DECIMAL(5, 2) DEFAULT 0,
	earned_leave DECIMAL(5, 2) DEFAULT 0,
	used_leave DECIMAL(5, 2) DEFAULT 0,
	available_leave DECIMAL(5, 2) DEFAULT 0,
	year INTEGER NOT NULL,
	last_updated TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
	UNIQUE(employee_id, leave_id, year),
	CONSTRAINT fk_employee FOREIGN KEY(employee_id) REFERENCES employees(id) ON DELETE CASCADE,
	CONSTRAINT fk_leave FOREIGN KEY(leave_id) REFERENCES leave_master(leave_id) ON DELETE CASCADE
);

-- Leave Accrual
CREATE TABLE leave_accrual (
	accrual_id BIGSERIAL PRIMARY KEY,
	employee_id BIGINT NOT NULL,
	leave_id BIGINT NOT NULL,
	credited_days DECIMAL(5, 2),
	accrual_date DATE,
	remarks TEXT,
	CONSTRAINT fk_employee FOREIGN KEY(employee_id) REFERENCES employees(id),
	CONSTRAINT fk_leave FOREIGN KEY(leave_id) REFERENCES leave_master(leave_id)
);

-- Leave Application
CREATE TABLE leave_applications (
	application_id BIGSERIAL PRIMARY KEY,
	employee_id BIGINT NOT NULL,
	leave_id BIGINT NOT NULL,
	from_date DATE NOT NULL,
	to_date DATE NOT NULL,
	total_days DECIMAL NOT NULL,
	reason TEXT,
	is_full_day BOOLEAN DEFAULT TRUE,
	status VARCHAR DEFAULT 'PENDING',
	applied_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
	approved_by BIGINT REFERENCES employees(id),
	approved_at TIMESTAMP,
	rejection_reason TEXT,
	CONSTRAINT fk_leave FOREIGN KEY(leave_id) REFERENCES leave_master(leave_id),
	CONSTRAINT fk_employee FOREIGN KEY(employee_id) REFERENCES employees(id)
);

-- Leave Approval History
CREATE TABLE leave_approval_history (
    approval_id BIGSERIAL PRIMARY KEY,
    application_id BIGINT NOT NULL,
    approver_id BIGINT REFERENCES employees(id),
    action VARCHAR(30),
    remarks TEXT,
    action_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
	CONSTRAINT fk_application FOREIGN KEY(application_id) REFERENCES leave_applications(application_id) ON DELETE CASCADE
);

-- Leave Cancellation
CREATE TABLE leave_cancellation (
    cancellation_id BIGSERIAL PRIMARY KEY,

    REFERENCES leave_applications(application_id),

-- Attendance Table
CREATE TABLE attendances(
	id BIGSERIAL PRIMARY KEY,
	employee_id BIGINT NOT NULL,
	attendance_date DATE NOT NULL,
	check_in TIMESTAMP,
	check_out TIMESTAMP,
	shift_id BIGINT,
	attendance_policy_id BIGINT,
	total_hours DECIMAL(5, 2),
	late_minutes INT DEFAULT 0,
	early_departure_minutes INT DEFAULT 0,
	overtime_hours DECIMAL(5, 2) DEFAULT 0,
	attendance_status VARCHAR(20) CHECK(attendance_status IN ('PRESENT', 'ABSENT', 'HALF_DAY', 'LEAVE', 'HOLIDAY', 'WEEK_OFF')),
	location VARCHAR(255),
	remarks TEXT,
	created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
	updated_at TIMESTAMP,
	UNIQUE(employee_id, attendance_date),
	CONSTRAINT fk_employee FOREIGN KEY(employee_id) REFERENCES employees(id),
	CONSTRAINT fk_attendance_policy FOREIGN KEY(attendance_policy_id) REFERENCES attendance_policies(policy_id),
	CONSTRAINT fk_shift FOREIGN KEY(shift_id) REFERENCES shifts(shift_id)
);

-- Attendance Regularizations
CREATE TABLE attendance_regularizations (
	regularization_id BIGSERIAL PRIMARY KEY,
	attendance_id BIGINT REFERENCES attendances(id),
	employee_id BIGINT REFERENCES employees(id),
	regularisation_date DATE,
	request_type VARCHAR(30) CHECK(
		request_type IN ('CHECK_IN', 'CHECK_OUT', 'BOTH')
	),
	reason TEXT,
	requested_check_in TIMESTAMP,
	requested_check_out TIMESTAMP,
	status VARCHAR(20) DEFAULT 'PENDING' CHECK (
		status IN ('PENDING', 'APPROVED', 'REJECTED')
	),
	approved_by BIGINT REFERENCES employees(id),
	rejected_reason TEXT,
	created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
	updated_at TIMESTAMP
);

-- Attendance regularization history
CREATE TABLE attendance_regularization_history (
	history_id BIGSERIAL PRIMARY KEY,
	regularization_id BIGINT NOT NULL,
	employee_id BIGINT NOT NULL,
	action VARCHAR(30) NOT NULL CHECK(action IN ('SUBMITTED', 'UPDATED', 'APPROVED', 'REJECTED', 'DELETED')),
	action_by BIGINT NOT NULL,
	remark TEXT,
	old_status VARCHAR(30),
	new_status VARCHAR(30),
	created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
	CONSTRAINT fk_regularization FOREIGN KEY(regularization_id) REFERENCES attendance_regularizations(regularization_id) ON DELETE CASCADE,
	CONSTRAINT fk_empoyee FOREIGN KEY(employee_id) REFERENCES employees(id),
	CONSTRAINT fk_action_by FOREIGN KEY(action_by) REFERENCES employees(id)
);
